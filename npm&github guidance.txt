npm
============================================
Frontend

create: => 

# npm 6.x (outdated, but still used by some):
npm create vite@latest introdemo --template react

# npm 7+, extra double-dash is needed:
npm create vite@latest introdemo -- --template react

cd introdemo
npm install

RUN => npm run dev

libraries: =>

npm install axios

npm install json-server --save-dev

============================================
Backend

npm init //create the project directory

npm start //start the server by running the js file specified in package.json scripts

node --watch index.js //automatic server changes tracking

npm run dev //write the previous command in package.json scripts as dev and use it to start the server in dev mode

============================================

GitHub

git status
git add .
git commit -m "message"
git push

======================
Back to an old commit|
======================

git log //detailed info for all commits on the repo
git reset --hard <commit_hash> //reset the repo to the commit linked to hash
git push origin HEAD --force //pushes the changes to github
